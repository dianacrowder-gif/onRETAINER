  <ul id="fileList"></ul>


  <script type="module">

    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-app.js";

    import { getStorage, ref, uploadBytes, getDownloadURL, listAll } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-storage.js";

    import { getAuth, onAuthStateChanged, signOut } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-auth.js";


    const firebaseConfig = {

      apiKey: "AIzaSyAf4jt3ecmL8Y6xsz27YaGStmtzjea9U",

      authDomain: "onretainer-d2396.firebaseapp.com",

      projectId: "onretainer-d2396",

      storageBucket: "onretainer-d2396.appspot.com",

      messagingSenderId: "30415789355",

      appId: "1:30415789355:web:8ccb4a93f8c3d79485796e "

    };


    const app = initializeApp(firebaseConfig);

    const auth = getAuth(app);

    const storage = getStorage(app);


    const fileInput = document.getElementById("fileInput");

    const uploadBtn = document.getElementById("uploadBtn");

    const fileList = document.getElementById("fileList");

    const logoutBtn = document.getElementById("logout");


    logoutBtn.addEventListener("click", () => {

      signOut(auth).then(() => {

        window.location.href = "index.html";

      });

    });


    uploadBtn.addEventListener("click", async () => {

      const file = fileInput.files[0];

      if (!file) return alert("Please choose a file first!");


      const user = auth.currentUser;

      if (!user) return alert("You must be logged in!");


      const fileRef = ref(storage, `uploads/${user.uid}/${file.name}`);

      await uploadBytes(fileRef, file);

      alert("Upload successful!");

      loadFiles();

    });


    async function loadFiles() {

      const user = auth.currentUser;

      if (!user) return;

      const userFolderRef = ref(storage, `uploads/${user.uid}`);

      const result = await listAll(userFolderRef);


      fileList.innerHTML = "";

      for (const itemRef of result.items) {

        const url = await getDownloadURL(itemRef);

        const li = document.createElement("li");

        li.innerHTML = `<a href="${url}" target="_blank">${itemRef.name}</a>`;

        fileList.appendChild(li);

      }

    }


    onAuthStateChanged(auth, (user) => {

      if (user) loadFiles();

      else window.location.href = "index.html";

    });

  </script>
